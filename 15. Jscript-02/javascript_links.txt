function fibonacciGenerator (n) {
//Do NOT change any of the code above ðŸ‘†
    
    var fib = [];
    //Write your code here:
    if( n ===1){
        return [0];
    }
    
    if (n===2){
        return [0,1];
    } 
    
    if (n>2){
        fib=[0,1];
    }
    
    for (var i =2; i< n; i++){
        
     var temp = fib[i-1] + fib[i-2];
     fib.push(temp);
    }
    
    
    //Return an array of fibonacci numbers starting from 0.
    return fib;

}
